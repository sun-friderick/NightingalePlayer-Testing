########################################################################################
#############               生成 base64 库文件                            ############## 
########################################################################################
LIST (APPEND base64_LIB_SRCS
    ${CMAKE_CURRENT_SOURCE_DIR}/Base64.cpp
    )
    
   
########################################################################################
#############              添加头文件目录                                 ############## 
########################################################################################
include_directories(  
    ${PROJECT_SOURCE_DIR}/includes  
    ${PROJECT_SOURCE_DIR}/Utils/Crypto
    ${PROJECT_SOURCE_DIR}/Utils/Crypto/Base64   
)  

########################################################################################
#############               添加 链接库文件目录                           ############## 
########################################################################################
link_directories(  
    /usr/local/lib   
    )  


    
########################################################################################
#############           生成目标库文件 并 标明版本号                      ############## 
########################################################################################
IF (MODULE_base64)
    #生成动态库  静态库则 STATIC  
    add_library (base64lib SHARED      ${base64_LIB_SRCS})  
    add_library (base64s STATIC        ${base64_LIB_SRCS})  
    # 设置版本号，SOVERSION为API版本号
    set_target_properties(base64lib   PROPERTIES VERSION 1.0.0 SOVERSION 1 )
ELSE (MODULE_base64)
    MESSAGE(STATUS "Not Include base64 module.")
ENDIF (MODULE_base64)
