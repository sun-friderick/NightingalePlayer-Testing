########################################################################################
#############               生成 sha1 库文件                            ############## 
########################################################################################
LIST (APPEND sha1_LIB_SRCS
    ${CMAKE_CURRENT_SOURCE_DIR}/SHA1.cpp
    )

    
########################################################################################
#############              添加头文件目录                                 ############## 
########################################################################################
include_directories(  
    ${PROJECT_SOURCE_DIR}/includes  
    ${PROJECT_SOURCE_DIR}/Utils/Crypto
    ${PROJECT_SOURCE_DIR}/Utils/Crypto/SHA1
)  

########################################################################################
#############               添加 链接库文件目录                           ############## 
########################################################################################
link_directories(  
    /usr/local/lib   
    )  


    
########################################################################################
#############           生成目标库文件 并 标明版本号                      ############## 
########################################################################################
IF (MODULE_sha1)
    #生成动态库  静态库则 STATIC  
    add_library (sha1lib SHARED      ${sha1_LIB_SRCS})  
    add_library (sha1s STATIC        ${sha1_LIB_SRCS})  
    # 设置版本号，SOVERSION为API版本号
    set_target_properties(sha1lib   PROPERTIES VERSION 1.0.0 SOVERSION 1 )
ELSE (MODULE_sha1)
    MESSAGE(STATUS "Not Include sha1 module.")
ENDIF (MODULE_sha1)
